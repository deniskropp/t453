// persistence.ts — Minimal NDJSON file-backed event store + snapshot support (optional fs)

import type { EcosystemEvent } from './events';
import type { RuntimeState } from './state';
import type { MetaDNA } from './metaDNA';
import { appendFile, readFile, writeFile, mkdir } from 'fs/promises';
import { existsSync } from 'fs';

export interface Snapshot {
  readonly version: '1.2';
  readonly timestamp: number;
  readonly state: RuntimeState;
  readonly lastCorrelationId: string;
  readonly metaDNA: MetaDNA;
  readonly eventCount: number;
}

export class FileEventStore {
  private readonly eventsPath: string;
  private readonly snapshotPath: string;
  private readonly snapshotInterval: number;

  constructor(
    basePath: string = './ecosystem-data',
    snapshotInterval = 25
  ) {
    this.eventsPath = `${basePath}/events.ndjson`;
    this.snapshotPath = `${basePath}/snapshot.json`;
    this.snapshotInterval = snapshotInterval;
  }

  private async ensureDir(): Promise<void> {
    const dir = this.eventsPath.substring(0, this.eventsPath.lastIndexOf('/'));
    if (!existsSync(dir)) {
      await mkdir(dir, { recursive: true });
    }
  }

  async saveEvent(event: EcosystemEvent): Promise<void> {
    await this.ensureDir();
    const line = JSON.stringify(event) + '\n';
    await appendFile(this.eventsPath, line, 'utf8');
  }

  async loadEvents(): Promise<EcosystemEvent[]> {
    if (!existsSync(this.eventsPath)) return [];
    const content = await readFile(this.eventsPath, 'utf8');
    return content
      .trim()
      .split('\n')
      .filter(Boolean)
      .map((line) => JSON.parse(line) as EcosystemEvent);
  }

  async saveSnapshot(
    state: RuntimeState,
    lastCorrelationId: string,
    eventCount: number,
    metaDNA: MetaDNA
  ): Promise<void> {
    await this.ensureDir();
    const snapshot: Snapshot = {
      version: '1.2',
      timestamp: Date.now(),
      state,
      lastCorrelationId,
      metaDNA,
      eventCount,
    };
    await writeFile(this.snapshotPath, JSON.stringify(snapshot, null, 2), 'utf8');
  }

  async loadLatestSnapshot(): Promise<Snapshot | null> {
    if (!existsSync(this.snapshotPath)) return null;
    try {
      const content = await readFile(this.snapshotPath, 'utf8');
      const snap = JSON.parse(content) as Snapshot;
      if (snap.version !== '1.2') {
        console.warn('[Persistence] Snapshot version mismatch — ignoring');
        return null;
      }
      return snap;
    } catch {
      return null;
    }
  }

  getSnapshotInterval(): number {
    return this.snapshotInterval;
  }
}
```

// runtime.ts — Extended with persistence, bootstrapFromDisk, auto-snapshot (v1.2 + durable store)

import type { EcosystemEvent, DecisionEvent } from './events';
import { initialState, reduce, loadScopedMemory, type RuntimeState } from './state';
import { handleArbitration } from './arbitration';
import { createMetricsEvent, computeConvergenceFromLog } from './monitor';
import { exportMetaDNA, type MetaDNA } from './metaDNA';
import { FileEventStore, type Snapshot } from './persistence';

export class EcosystemRuntime {
  private eventLog: EcosystemEvent[] = [];
  private state: RuntimeState = initialState();
  private store?: FileEventStore;

  constructor(storePath?: string, snapshotInterval = 25) {
    if (storePath) {
      this.store = new FileEventStore(storePath, snapshotInterval);
    }
  }

  get currentState(): Readonly<RuntimeState> {
    return this.state;
  }

  get eventLogSnapshot(): ReadonlyArray<EcosystemEvent> {
    return [...this.eventLog];
  }

  async append(event: EcosystemEvent): Promise<void> {
    this.eventLog.push(event);
    this.state = reduce(this.state, event);

    if (this.store) {
      await this.store.saveEvent(event);

      // Auto-snapshot every N events
      if (this.eventLog.length % this.store.getSnapshotInterval() === 0) {
        const lastCorr = event.correlation_id;
        await this.store.saveSnapshot(
          this.state,
          lastCorr,
          this.eventLog.length,
          exportMetaDNA()
        );
      }
    }
  }

  // Full in-memory replay (original behavior)
  replay(): RuntimeState {
    this.state = initialState();
    for (const event of this.eventLog) {
      this.state = reduce(this.state, event);
    }
    return this.state;
  }

  // NEW: bootstrapFromDisk — prefers snapshot + tail replay for speed
  async bootstrapFromDisk(): Promise<RuntimeState> {
    if (!this.store) {
      console.warn('[Runtime] No persistence store configured — bootstrap skipped');
      return this.state;
    }

    const snap = await this.store.loadLatestSnapshot();
    const allEvents = await this.store.loadEvents();

    if (snap) {
      // Restore from snapshot
      this.state = { ...snap.state };
      this.eventLog = allEvents; // keep full log for audit

      // Replay only tail events after snapshot timestamp
      const tailEvents = allEvents.filter(
        (e) => e.timestamp > snap.timestamp || e.correlation_id !== snap.lastCorrelationId
      );
      for (const event of tailEvents) {
        this.state = reduce(this.state, event);
      }
      console.log(
        `[Runtime] Bootstrapped from snapshot @ ${new Date(snap.timestamp).toISOString()} + ${tailEvents.length} tail events`
      );
    } else if (allEvents.length > 0) {
      // No snapshot — full replay from disk
      this.eventLog = allEvents;
      this.state = initialState();
      for (const event of allEvents) {
        this.state = reduce(this.state, event);
      }
      console.log(`[Runtime] Bootstrapped from full event log (${allEvents.length} events)`);
    }

    return this.state;
  }

  async saveManualSnapshot(): Promise<void> {
    if (!this.store) return;
    const lastCorr = this.eventLog.length > 0
      ? this.eventLog[this.eventLog.length - 1].correlation_id
      : 'none';
    await this.store.saveSnapshot(this.state, lastCorr, this.eventLog.length, exportMetaDNA());
  }

  // ... (all previous methods: hydrateMemory, maybeArbitrate, emitCycleComplete, exportMetaDNA, simulateCycle remain unchanged)
  // hydrateMemory, maybeArbitrate, emitCycleComplete, etc. — identical to previous version
}
```

⫻context/klmx:DesignNotes/1
- NDJSON chosen for human-readable append simplicity and easy tailing/grepping.
- Snapshot contains full RuntimeState + lastCorrelationId + Meta-DNA (version + checksum via exported fields) for safe evolution detection.
- bootstrapFromDisk prefers snapshot for speed on long histories; falls back to full replay if no snapshot.
- fs/promises is dynamically usable — if you run in a non-Node environment the store simply stays undefined and everything remains pure in-memory.
- No breaking changes to existing in-memory API; persistence is additive and opt-in.
- Audit trail is complete: every event, every snapshot, every Meta-DNA version is on disk.
