export type EventType =
  | 'decision'
  | 'metrics'
  | 'memory_write'
  | 'arbitration'
  | 'visual';

export interface BaseEvent {
  readonly type: EventType;
  readonly timestamp: number;           // unix ms
  readonly source_stage: string;
  readonly correlation_id: string;      // for replay & audit
}

export interface DecisionPayload {
  consent: 'granted' | 'conditional' | 'denied';
  scope: string[];                      // runtime-enforceable operational guard
  expiry?: number;                      // cycle count or absolute timestamp
  rationale: string;
  arbitration_required: boolean;
}

export interface DecisionEvent extends BaseEvent {
  readonly type: 'decision';
  readonly payload: DecisionPayload;
}

export interface MetricsPayload {
  coherence_score: number;   // 0-1
  novelty_delta: number;
  consent_stability: number;
  cycle: number;
}

export interface MetricsEvent extends BaseEvent {
  readonly type: 'metrics';
  readonly payload: MetricsPayload;
}

export type MemoryType = 'episodic' | 'semantic' | 'policy';

export type WritePolicy =
  | 'append-only'      // episodic
  | 'validated-merge'  // semantic
  | 'immutable-log';   // policy

export interface MemoryWritePayload {
  memory_type: MemoryType;
  write_policy: WritePolicy;
  content: unknown;
}

export interface MemoryWriteEvent extends BaseEvent {
  readonly type: 'memory_write';
  readonly payload: MemoryWritePayload;
}

export interface ArbitrationPayload {
  strategy: 'balanced' | 'guard-dominant' | 'flow-dominant' | 'human-escalation';
  decision_correlation_id: string;
  outcome: 'resolved' | 'escalated' | 'passed-through';
  rationale: string;
}

export interface ArbitrationEvent extends BaseEvent {
  readonly type: 'arbitration';
  readonly payload: ArbitrationPayload;
}

export interface VisualPayload {
  description: string;
  data: unknown;           // e.g. diagram spec, image ref, or KickLang visual block
  linked_to: 'state' | 'insight' | 'both';
}

export interface VisualEvent extends BaseEvent {
  readonly type: 'visual';
  readonly payload: VisualPayload;
}

export type EcosystemEvent =
  | DecisionEvent
  | MetricsEvent
  | MemoryWriteEvent
  | ArbitrationEvent
  | VisualEvent;

// state.ts — Derived state, reducers, and scoped memory hydration (v1.2)

import type {
  EcosystemEvent,
  DecisionEvent,
  MetricsEvent,
  MemoryWriteEvent,
  MemoryType,
  WritePolicy,
} from './events';

export interface Memory {
  episodic: unknown[];
  semantic: unknown[];
  policy: unknown[];
}

export interface RuntimeState {
  memory: Memory;
  lastDecision?: DecisionEvent['payload'];
  currentMetrics?: MetricsEvent['payload'];
  cycle: number;
  convergenceReached: boolean;
}

export const initialState = (): RuntimeState => ({
  memory: { episodic: [], semantic: [], policy: [] },
  cycle: 0,
  convergenceReached: false,
});

function applyWritePolicy(
  current: unknown[],
  content: unknown,
  policy: WritePolicy
): unknown[] {
  switch (policy) {
    case 'append-only':
      return [...current, content];
    case 'validated-merge':
      // Simple dedup + novelty guard (extend with real coherence check)
      if (current.some((item) => JSON.stringify(item) === JSON.stringify(content))) {
        return current;
      }
      return [...current, content];
    case 'immutable-log':
      return [...current, { ...content, logged_at: Date.now() }];
    default:
      return current;
  }
}

export function reduce(
  state: RuntimeState,
  event: EcosystemEvent
): RuntimeState {
  const next = { ...state, cycle: state.cycle + 1 };

  switch (event.type) {
    case 'decision':
      next.lastDecision = event.payload;
      return next;

    case 'metrics':
      next.currentMetrics = event.payload;
      // deterministic convergence from log
      next.convergenceReached =
        event.payload.coherence_score > 0.9 &&
        event.payload.novelty_delta < 0.05 &&
        event.payload.consent_stability > 0.95;
      return next;

    case 'memory_write': {
      const { memory_type, write_policy, content } = event.payload;
      const mem = { ...next.memory };
      mem[memory_type] = applyWritePolicy(mem[memory_type], content, write_policy);
      next.memory = mem;
      return next;
    }

    case 'arbitration':
    case 'visual':
      // side-effect free for now; extend with visual/state linking if needed
      return next;

    default:
      return next;
  }
}

// Scope-aware memory slice loader (Stage00)
export function loadScopedMemory(
  fullMemory: Memory,
  intent: string,
  scope: string[]
): Memory {
  // Minimal filter: keep items whose tags or content mention intent/scope
  // In production: use embeddings or explicit tagging
  const filter = (arr: unknown[]) =>
    arr.filter((item: any) =>
      scope.some((s) => JSON.stringify(item).toLowerCase().includes(s.toLowerCase())) ||
      (item?.tags && item.tags.some((t: string) => scope.includes(t)))
    );

  return {
    episodic: filter(fullMemory.episodic),
    semantic: filter(fullMemory.semantic),
    policy: filter(fullMemory.policy),
  };
}

import type { DecisionEvent, ArbitrationEvent, ArbitrationPayload } from './events';

export interface ArbitrationResult {
  event?: ArbitrationEvent;
  outcome: 'resolved' | 'escalated' | 'passed-through';
  rationale: string;
}

export function handleArbitration(
  decision: DecisionEvent,
  strategy: ArbitrationPayload['strategy'] = 'balanced'
): ArbitrationResult {
  if (!decision.payload.arbitration_required) {
    return {
      outcome: 'passed-through',
      rationale: 'No conflict detected — normal agent autonomy preserved',
    };
  }

  // Placeholder logic — replace with real multi-agent conflict rules
  const rationale = `Arbitration executed with strategy=${strategy} for scope=${decision.payload.scope.join(',')}`;
  const outcome: ArbitrationResult['outcome'] =
    strategy === 'guard-dominant' ? 'resolved' : 'escalated';

  const arbEvent: ArbitrationEvent = {
    type: 'arbitration',
    timestamp: Date.now(),
    source_stage: 'Stage16',
    correlation_id: decision.correlation_id,
    payload: {
      strategy,
      decision_correlation_id: decision.correlation_id,
      outcome,
      rationale,
    },
  };

  return { event: arbEvent, outcome, rationale };
}

import type { MetricsEvent, MetricsPayload, EcosystemEvent } from './events';

export function createMetricsEvent(
  payload: MetricsPayload,
  source_stage = 'Stage20',
  correlation_id = crypto.randomUUID()
): MetricsEvent {
  return {
    type: 'metrics',
    timestamp: Date.now(),
    source_stage,
    correlation_id,
    payload,
  };
}

export function computeConvergenceFromLog(log: EcosystemEvent[]): boolean {
  const metricsEvents = log.filter((e): e is MetricsEvent => e.type === 'metrics');
  if (metricsEvents.length === 0) return false;

  const latest = metricsEvents[metricsEvents.length - 1].payload;
  return (
    latest.coherence_score > 0.9 &&
    latest.novelty_delta < 0.05 &&
    latest.consent_stability > 0.95
  );
}

// runtime.ts — Core event-sourced orchestrator with replay & Stage20 recursion hook

import type { EcosystemEvent, DecisionEvent } from './events';
import { initialState, reduce, loadScopedMemory, type RuntimeState } from './state';
import { handleArbitration } from './arbitration';
import { createMetricsEvent, computeConvergenceFromLog } from './monitor';
import { exportMetaDNA, type MetaDNA } from './metaDNA';

export class EcosystemRuntime {
  private eventLog: EcosystemEvent[] = [];
  private state: RuntimeState = initialState();

  get currentState(): Readonly<RuntimeState> {
    return this.state;
  }

  get eventLogSnapshot(): ReadonlyArray<EcosystemEvent> {
    return [...this.eventLog];
  }

  append(event: EcosystemEvent): void {
    this.eventLog.push(event);
    this.state = reduce(this.state, event);
  }

  // Full replay from empty state (event sourcing guarantee)
  replay(): RuntimeState {
    this.state = initialState();
    for (const event of this.eventLog) {
      this.state = reduce(this.state, event);
    }
    return this.state;
  }

  // Stage00-style scoped memory load
  hydrateMemory(intent: string, scope: string[]): RuntimeState {
    const scoped = loadScopedMemory(this.state.memory, intent, scope);
    this.state = { ...this.state, memory: scoped };
    return this.state;
  }

  // Stage16 conflict-only arbitration
  maybeArbitrate(decision: DecisionEvent, strategy?: any) {
    const result = handleArbitration(decision, strategy);
    if (result.event) {
      this.append(result.event);
    }
    return result;
  }

  // Stage20: emit metrics + three policy-enforced memory writes + recursion trigger
  emitCycleComplete(
    metricsPayload: Parameters<typeof createMetricsEvent>[0],
    memoryWrites: Array<{ memory_type: any; write_policy: any; content: unknown }>
  ): void {
    // metrics event
    const metricsEvent = createMetricsEvent(metricsPayload);
    this.append(metricsEvent);

    // three distinct memory writes (policies enforced in reducer)
    for (const write of memoryWrites) {
      this.append({
        type: 'memory_write',
        timestamp: Date.now(),
        source_stage: 'Stage20',
        correlation_id: metricsEvent.correlation_id,
        payload: write as any,
      });
    }

    // convergence check from log
    this.state.convergenceReached = computeConvergenceFromLog(this.eventLog);
  }

  exportMetaDNA(): MetaDNA {
    return exportMetaDNA();
  }

  // Convenience: simulate a full high-level cycle (for testing/demo)
  simulateCycle(userRequest: string, intent: string, scope: string[]) {

    console.log(`[Runtime] Simulating cycle for: ${userRequest}`);
    this.hydrateMemory(intent, scope);
    // ... further stage orchestration would live here or in a higher coordinator

}
